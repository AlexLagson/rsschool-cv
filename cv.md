 ``` Javascript
function isPrime(num) {
    let _true = 'Number is prime',
     _false = 'wrong answer'; 
    if (num > 1000) {
        console.log('Incorrect data');}
    if (num <= 1) return _false;
    if (num == 2) return _true;
    for (let i = 2; i <= num / 2; i++) {
    if (num % i == 0) {
      return _false;
    }
    }
    return _true;
    }
 ```
